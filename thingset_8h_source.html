126 /*
127  * Internal access rights to data objects
128  */
129 #define TS_READ_ALL (0x1U << 0) // read access for all
130 #define TS_READ_USER (0x3U << 0) // read after authentication as normal user
131 #define TS_READ_MAKER (0x7U << 0) // read after authentication as manufacturer, e.g. for factory calibration
132 #define TS_READ_MASK TS_READ_MAKER // maker has all 3 bits set
133 
134 #define TS_WRITE_ALL (0x1U << 3) // write access for all
135 #define TS_WRITE_USER (0x3U << 3) // write after authentication as normal user
136 #define TS_WRITE_MAKER (0x7U << 3) // write after authentication as manufacturer, e.g. for factory calibration
137 #define TS_WRITE_MASK TS_WRITE_MAKER
138 
139 #define TS_EXEC_ALL (0x1U << 6) // execute access for all (only for RPC)
140 #define TS_EXEC_USER (0x3U << 6) // execute after authentication as normal user
141 #define TS_EXEC_MAKER (0x7U << 6) // execute after authentication as manufacturer, e.g. for factory calibration
142 #define TS_EXEC_MASK TS_EXEC_MAKER
